## 1. Clase ProductoPanaderia.java

```java
import java.time.LocalDate;

public class ProductoPanaderia {
    private String nombre;
    private String tipo; // "pan", "pastel", "galleta", "bollo", "tarta"
    private double precio;
    private int unidadesStock;
    private LocalDate fechaFabricacion;
    private boolean sinGluten;
    private double pesoGramos;
    private int caloriasPorcion;
    private boolean vegano;
    private String proveedor;

    public ProductoPanaderia(String nombre, String tipo, double precio, int unidadesStock,
                            LocalDate fechaFabricacion, boolean sinGluten, double pesoGramos,
                            int caloriasPorcion, boolean vegano, String proveedor) {
        this.nombre = nombre;
        this.tipo = tipo;
        this.precio = precio;
        this.unidadesStock = unidadesStock;
        this.fechaFabricacion = fechaFabricacion;
        this.sinGluten = sinGluten;
        this.pesoGramos = pesoGramos;
        this.caloriasPorcion = caloriasPorcion;
        this.vegano = vegano;
        this.proveedor = proveedor;
    }

    // Getters
    public String getNombre() { return nombre; }
    public String getTipo() { return tipo; }
    public double getPrecio() { return precio; }
    public int getUnidadesStock() { return unidadesStock; }
    public LocalDate getFechaFabricacion() { return fechaFabricacion; }
    public boolean isSinGluten() { return sinGluten; }
    public double getPesoGramos() { return pesoGramos; }
    public int getCaloriasPorcion() { return caloriasPorcion; }
    public boolean isVegano() { return vegano; }
    public String getProveedor() { return proveedor; }

    @Override
    public String toString() {
        return String.format("%-20s %-10s $%-6.2f Stock:%-3d %s",
                nombre, tipo, precio, unidadesStock,
                sinGluten ? "Sin gluten" : "");
    }
}
```

## 3. Clase adicional con consultas más complejas

```java
import java.util.DoubleSummaryStatistics;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

public class ConsultasAvanzadasPanaderia {

    public static void main(String[] args) {
        List<ProductoPanaderia> productos = PanaderiaStreams.getProductos();

        System.out.println("=== CONSULTAS AVANZADAS CON STREAMS ===");
        System.out.println();

        // 1. Precio promedio por tipo de producto
        System.out.println("1. Precio promedio por tipo:");
        Map<String, Double> precioPromedioPorTipo = productos.stream()
                .collect(Collectors.groupingBy(
                    ProductoPanaderia::getTipo,
                    Collectors.averagingDouble(ProductoPanaderia::getPrecio)
                ));

        precioPromedioPorTipo.forEach((tipo, promedio) ->
            System.out.printf("%-10s: $%.2f%n", tipo, promedio)
        );
        System.out.println();

        // 2. Producto más caro
        System.out.println("2. Producto más caro:");
        Optional<ProductoPanaderia> productoMasCaro = productos.stream()
                .max(java.util.Comparator.comparingDouble(ProductoPanaderia::getPrecio));

        productoMasCaro.ifPresent(p ->
            System.out.println(p.getNombre() + " - $" + p.getPrecio())
        );
        System.out.println();

        // 3. Estadísticas de precios
        System.out.println("3. Estadísticas de precios:");
        DoubleSummaryStatistics stats = productos.stream()
                .mapToDouble(ProductoPanaderia::getPrecio)
                .summaryStatistics();

        System.out.printf("""
                Total productos: %d
                Precio mínimo: $%.2f
                Precio máximo: $%.2f
                Precio promedio: $%.2f
                Suma total: $%.2f
                """,
                stats.getCount(),
                stats.getMin(),
                stats.getMax(),
                stats.getAverage(),
                stats.getSum()
        );
        System.out.println();

        // 4. Productos agrupados por proveedor
        System.out.println("4. Productos por proveedor:");
        Map<String, List<ProductoPanaderia>> productosPorProveedor = productos.stream()
                .collect(Collectors.groupingBy(ProductoPanaderia::getProveedor));

        productosPorProveedor.forEach((proveedor, listaProductos) -> {
            System.out.println("\n" + proveedor + ":");
            listaProductos.forEach(p -> System.out.println("  - " + p.getNombre()));
        });
        System.out.println();

        // 5. Buscar productos por nombre (parcial)
        System.out.println("5. Productos que contienen 'Tarta':");
        productos.stream()
                .filter(p -> p.getNombre().toLowerCase().contains("tarta"))
                .forEach(p -> System.out.println("  " + p.getNombre()));
    }
}
```

## 4. Clase para prácticas adicionales

```java
import java.util.List;
import java.util.stream.Collectors;

public class PracticaFiltersPanaderia {

    public static void ejerciciosPractica(List<ProductoPanaderia> productos) {
        System.out.println("=== EJERCICIOS DE PRÁCTICA ===");

        // Ejercicio 1: Obtener nombres de productos baratos (< 2€)
        System.out.println("\n1. Nombres de productos baratos (< 2€):");
        List<String> nombresBaratos = productos.stream()
                .filter(p -> p.getPrecio() < 2)
                .map(ProductoPanaderia::getNombre)
                .collect(Collectors.toList());
        nombresBaratos.forEach(System.out::println);

        // Ejercicio 2: Productos con buen stock (> 15 unidades)
        System.out.println("\n2. Productos con buen stock:");
        productos.stream()
                .filter(p -> p.getUnidadesStock() > 15)
                .map(p -> p.getNombre() + " (" + p.getUnidadesStock() + " unidades)")
                .forEach(System.out::println);

        // Ejercicio 3: Productos saludables (sin gluten y < 200 calorías)
        System.out.println("\n3. Productos saludables:");
        productos.stream()
                .filter(p -> p.isSinGluten() && p.getCaloriasPorcion() < 200)
                .map(p -> p.getNombre() + " - " + p.getCaloriasPorcion() + " cal")
                .forEach(System.out::println);

        // Ejercicio 4: Tipos de productos únicos
        System.out.println("\n4. Tipos de productos disponibles:");
        productos.stream()
                .map(ProductoPanaderia::getTipo)
                .distinct()
                .forEach(System.out::println);
    }
}
```
